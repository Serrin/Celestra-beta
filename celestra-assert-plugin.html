Autotests

Functions and properties

Name Description
assert.strict = true Default value is true. If this property is true, then always the strict equality is used in these functions:
  • assert.equal();
  • assert.notEqual();
  • assert.deepEqual();
  • assert.notDeepEqual();
assert(condition [, message | error]); Same as
celestra.assert(condition [, message | error]);.
assert.deepEqual(value1, value2 [, message | error]); Same as
celestra.deepStrictEqual(value1, value2 [, message | error]);
or
celestra.assertDeepEqual(value1, value2 [, message | error]);.
assert.doesNotMatch(string, regexp, message); Same as
celestra.assertDoesNotMatch(string, regexp, message);.
assert.equal(value1, value2 [, message | error]); Same as
celestra.assertStrictEqual(value1, value2 [, message | error]);
or
celestra.assertEqual(value1, value2 [, message | error]);.
assert.fail(message | error); Same as
celestra.assertFail(message | error);.
assert.is(value, expectedType [, message | error]); Same as
celestra.assertIs(value, expectedType [, message | error]);.
assert.isFalse(condition [, message | error]); Same as
celestra.assertFalse(condition [, message | error]);.
assert.isNil(value [, message | error]); Same as
celestra.assertIsNil(value [, message | error]);.
assert.isNot(value, expectedType [, message | error]); Same as
celestra.assertIsNot(value, expectedType [, message | error]);.
assert.isNotNil(value [, message | error]); Same as
celestra.assertIsNotNil(value [, message | error]);.
assert.isTrue(condition [, message | error]); Same as
celestra.assertTrue(condition [, message | error]);.
assert.match(string, regexp, message); Same as
celestra.assertMatch(string,regexp, message);.
assert.notDeepEqual(value1, value2 [, message | error]); Same as
celestra.assertNotDeepStrictEqual(value1, value2 [, message | error]);
or
celestra.assertNotDeepEqual(value1, value2 [, message | error]);.
assert.notEqual(value1, value2 [, message | error]); Same as
celestra.assertNotStrictEqual(value1, value2 [, message | error]);
or
celestra.assertNotEqual(value1, value2 [, message | error]);.
assert.notStrictEqual(value1, value2 [, message | error]); Same as
celestra.assertNotStrictEqual(value1, value2 [, message | error]);.
assert.ok(condition [, message | error]); Same as
celestra.assert(condition [, message | error]);.
assert.strictEqual(value1, value2 [, message | error]); Same as
celestra.assertStrictEqual(value1, value2 [, message | error]);.
assert.Throws(callback [, message | error]); Same as
celestra.assertThrows(callback [, message | error]);.
